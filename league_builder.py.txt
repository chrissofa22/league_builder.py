import csv

def SDR(thing):
    with open("teams.txt", "a") as file:
        file.write(thing+"\n")
        

def show():
    with open("teams.txt") as file:
        for line in file:
            print(line)    
    
    
    
    
if __name__ == '__main__':
    if csv.argv[1].lower() == '--list':
        show()
    else:
        SDR(' '.join(csv.argv[1:]))
